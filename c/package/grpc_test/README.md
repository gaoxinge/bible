```
$ cd external
$ git clone git@github.com:grpc/grpc.git --depth=1

$ cd grpc
$ git submodule update --init
$ mkdir build/
$ mkdir install/
$ cd build/
$ cmake .. -G "Visual Studio 16 2019"
$ cmake --build . --config Release
$ cmake --install . --prefix ../install
```

```
$ cd examples/cpp/helloworld
$ cmake -B build0/ -DCMAKE_PREFIX_PATH=../../../install/
$ cmake --build build0/ --config Release
```
